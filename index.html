function procesarAjusteOido(points, ear) { // 1. Filtrar puntos válidos y ORDENARLOS por intensidad (dB) de forma ascendente let validos = points.filter(p => p.db !== null && p.pct !== null && !isNaN(p.db) && !isNaN(p.pct)); if (validos.length < 2) return null; validos.sort((a, b) => a.db - b.db); let x_reales = validos.map(p => p.db); let y_reales = validos.map(p => p.pct); let prefix = ear === 'OD' ? 'od' : 'oi'; let va500 = document.getElementById(prefix + 'VA500').value; let va1000 = document.getElementById(prefix + 'VA1000').value; let va2000 = document.getElementById(prefix + 'VA2000').value; let va4000 = document.getElementById(prefix + 'VA4000').value; let vo500 = document.getElementById(prefix + 'VO500').value; let vo1000 = document.getElementById(prefix + 'VO1000').value; let vo2000 = document.getElementById(prefix + 'VO2000').value; let vo4000 = document.getElementById(prefix + 'VO4000').value; let ptaVA = (va500!==""&&va1000!==""&&va2000!==""&&va4000!=="") ? (parseFloat(va500)+parseFloat(va1000)+parseFloat(va2000)+parseFloat(va4000))/4 : null; let ptaVO = (vo500!==""&&vo1000!==""&&vo2000!==""&&vo4000!=="") ? (parseFloat(vo500)+parseFloat(vo1000)+parseFloat(vo2000)+parseFloat(vo4000))/4 : null; let pb_max = Math.max(...y_reales); let idx_max = y_reales.indexOf(pb_max); let pb_min = pb_max; if(idx_max !== -1 && idx_max < y_reales.length - 1) { pb_min = Math.min(...y_reales.slice(idx_max + 1)); } let rollover_index = pb_max > 0 ? (pb_max - pb_min) / pb_max : 0; let es_patologico = (rollover_index >= 0.45); let es_sospecha_leve = (rollover_index >= 0.35 && rollover_index < 0.45); let indice_corte = x_reales.length; if (es_patologico || es_sospecha_leve) { for (let i = idx_max + 1; i < y_reales.length; i++) { if (y_reales[i] < (pb_max * 0.8)) { indice_corte = i; break; } } } let x_filtrados = x_reales.slice(0, indice_corte); let y_filtrados = y_reales.slice(0, indice_corte); if (x_filtrados.length < 2) return null; let y_max_fijo = Math.max(...y_filtrados); if(y_max_fijo < 20) y_max_fijo = 100; // Ajuste matemático con pesos corregidos (escala logarítmica razonable 1:10:20) let mejorError = Infinity; let bestB = 0.1, bestX50 = 0; for (let b = 0.05; b <= 2.5; b += 0.05) { for (let x50 = -10; x50 <= 110; x50 += 0.5) { let sumaError = 0; for (let k = 0; k < x_filtrados.length; k++) { let y_teorica = y_max_fijo / (1 + Math.exp(-b * (x_filtrados[k] - x50))); let dif = y_filtrados[k] - y_teorica; let peso = 1; if (y_filtrados[k] >= 40 && y_filtrados[k] <= 60) peso = 10; if (y_filtrados[k] >= (y_max_fijo - 5)) peso = 20; sumaError += (dif * dif) * peso; } if (sumaError < mejorError) { mejorError = sumaError; bestB = b; bestX50 = x50; } } } let finalErrorSum = 0; let meanY = y_filtrados.reduce((a, b) => a + b, 0) / y_filtrados.length; let sumSquaresTotal = 0; for (let k = 0; k < x_filtrados.length; k++) { let y_teorica = y_max_fijo / (1 + Math.exp(-bestB * (x_filtrados[k] - bestX50))); finalErrorSum += Math.pow(y_filtrados[k] - y_teorica, 2); sumSquaresTotal += Math.pow(y_filtrados[k] - meanY, 2); } let r2_num = sumSquaresTotal === 0 ? 1 : 1 - (finalErrorSum / sumSquaresTotal); let p_max_calc = (y_max_fijo * bestB) / 4; // CONTROL DE CONTROL: Evitar asíntotas y colapsos en el cálculo del SRT al 50% let urvText = "No Alc. (<50%)"; let numericSRT = null; if (y_max_fijo > 50.1) { numericSRT = bestX50 - Math.log((y_max_fijo / 50) - 1) / bestB; urvText = `${numericSRT.toFixed(1)} dB HL`; } else if (y_max_fijo >= 50) { numericSRT = bestX50; // Salvavidas si es exactamente 50% urvText = `${numericSRT.toFixed(1)} dB HL`; } let rolloverText = "Negativo (<0.35)"; if (rollover_index >= 0.45) rolloverText = `⚠️ Significativo (${rollover_index.toFixed(2)})`; else if (rollover_index >= 0.35) rolloverText = `⚠️ Sospecha leve (${rollover_index.toFixed(2)})`; let sdt_val = document.getElementById(prefix + 'UDV').value; let mcl_val = document.getElementById(prefix + 'MCL').value; let ucl_val = document.getElementById(prefix + 'UCL').value; let sdtText = "--", rdText = "--", mclText = "--", reclutText = "--"; // Usar SRT si existe, si no, usar SDT (UDV) como referencia secundaria para no romper la tabla let umbralReferencia = numericSRT !== null ? numericSRT : (sdt_val !== "" ? parseFloat(sdt_val) : null); if (numericSRT !== null && sdt_val !== "") { let diffSDT = numericSRT - parseFloat(sdt_val); sdtText = parseFloat(sdt_val) >= numericSRT ? `❌ Incongruente` : (diffSDT > 15 ? `⚠️ Desfasado` : `🟢 Coherente`); } if (umbralReferencia !== null) { let labelRef = numericSRT !== null ? "SRT" : "SDT"; if (ucl_val !== "") { let rd = parseFloat(ucl_val) - umbralReferencia; rdText = rd < 45 ? `⚠️ ${rd.toFixed(1)} dB (${labelRef})` : `🟢 ${rd.toFixed(1)} dB`; reclutText = rd < 45 ? `⚠️ Positivo` : `🟢 Negativo`; } if (mcl_val !== "") { let mc = parseFloat(mcl_val) - umbralReferencia; mclText = mc < 20 ? `⚠️ ${mc.toFixed(1)} dB (${labelRef})` : `🟢 ${mc.toFixed(1)} dB`; } } let concordanciaText = "--"; let bloqueo_SRT = false; if (numericSRT !== null && ptaVA !== null) { let diffPTA_SRT = Math.abs(ptaVA - numericSRT); if (diffPTA_SRT <= 10) concordanciaText = `Congruente (${diffPTA_SRT.toFixed(1)} dB)`; else if (diffPTA_SRT <= 14) concordanciaText = `⚠️ Discrepancia leve (${diffPTA_SRT.toFixed(1)} dB)`; else { concordanciaText = `❌ Inconsistencia (${diffPTA_SRT.toFixed(1)} dB)`; bloqueo_SRT = true; } } let clasificacion = "Compatible con normalidad"; if (!bloqueo_SRT) { if (es_patologico) clasificacion = "⚠️ Sospecha de afectación neural (Roll-over significativo)"; else if (es_sospecha_leve) clasificacion = "⚠️ Zona gris: Sospecha leve de fatiga neural"; else if (numericSRT !== null) { if (y_max_fijo >= 90) { if (ptaVA !== null && ptaVO !== null) { clasificacion = (ptaVA - ptaVO >= 15) ? "Compatible con perfil Conductivo" : "Hipoacusia Neurosensorial/Coclear con excelente discriminación"; } else { clasificacion = "Perfil desplazado con WRS óptimo (Requiere valorar GAP tonal)"; } } else if (y_max_fijo >= 80) clasificacion = "WRS discretamente reducida (Compatible con presbiacusia o lesión inicial)"; else if (y_max_fijo >= 50) clasificacion = "WRS reducida (Compatible con distorsión coclear)"; else clasificacion = "WRS muy reducida (Valorar severidad coclear o desproporción neural)"; } else { clasificacion = "Discriminación máxima insuficiente para calcular SRT de forma estándar."; } } else { clasificacion = "⚠️ Inconsistencia logo-tonal significativa (SRT-PTA ≥ 15 dB)."; } if (r2_num < 0.85) clasificacion = "⚠️ Baja fiabilidad matemática (R² < 0.85)."; return { urv: urvText, urvVal: numericSRT, pbMax: `${y_max_fijo.toFixed(0)}%`, pendiente: `${p_max_calc.toFixed(1)} %/dB`, r2: r2_num.toFixed(3), r2Val: r2_num, rollover: rolloverText, sdtText: sdtText, rdText: rdText, mclText: mclText, reclutText: reclutText, concordancia: concordanciaText, clasif: clasificacion, b: bestB, x50: bestX50, yMax: y_max_fijo }; }
